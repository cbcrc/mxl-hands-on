# Declare build arguments for the base image
ARG BASE_IMAGE=debian:trixie-slim

FROM ${BASE_IMAGE}

# Declare build arguments after FROM
ARG BUILD_DIR=../dmf-mxl/build/Linux-GCC-Release_x86_64 

WORKDIR /app

# Use minimal dependencies and clean up in the same layer to reduce image size
RUN apt-get update && apt-get -y install --no-install-recommends \
        gstreamer1.0-plugins-base \
        gstreamer1.0-plugins-good \
        gstreamer1.0-plugins-bad \
        gstreamer1.0-plugins-ugly \
        gstreamer1.0-x \
        gstreamer1.0-tools \
        patchelf \
    && rm -rf /var/lib/apt/lists/*

# Copy only necessary files
COPY ./build-images/city.ts /app/
COPY ${BUILD_DIR}/utils/gst-looping-filesrc /gst-plugins/
COPY ${BUILD_DIR}/lib/*.so* /app/
COPY ${BUILD_DIR}/lib/tests/data/*.json /app/
COPY ${BUILD_DIR}/tools/mxl-gst/mxl-gst-looping-filesrc /app/

# Set environment variable for GStreamer plugin path
ENV GST_PLUGIN_PATH="/gst-plugins"

# Set rpath for the executable
RUN patchelf --set-rpath /app mxl-gst-looping-filesrc

CMD ["/app/mxl-gst-looping-filesrc", "-d", "/domain","-i", "city.ts"]
